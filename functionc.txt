#include <stdio.h>

int getValueFromArray(int *array, int len, int index);

int main(void)
{
    int array[] = {'a', 'b', 'c'};
    int len = sizeof(array) / sizeof(int);

    for (int i = 0; i < len; i++)
    {
        int value = getValueFromArray(array, len, i);
        if (value != -1)
        {
            printf("Element at index %d is %c\n", i, value);
        }
    }
    return 0;
}

int getValueFromArray(int *array, int len, int index)

{

    if (index < 0 || index>= len)
{
    //index would be out of bounds
    return -1;
}
    return array[index];

}
